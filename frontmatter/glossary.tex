\newglossaryentry{plm}{
    name={Protein Language Model},
    text={protein language model},
    description={is a language model that is pretrained on large-scale unlabeled protein sequences. It captures the statistical properties of protein sequences and can be fine-tuned on downstream tasks}
}

\newglossaryentry{deep-learning}{
    name={Deep Learning},
    text={deep learning},
    description={is a subfield of machine learning that uses neural networks with many layers to model and make sense of data}
}

\newglossaryentry{ensemble-learning}{
    name={Ensemble Learning},
    text={ensemble learning},
    description={is a machine learning technique that combines several base models to produce one optimal predictive model by voting or averaging}
}

\newglossaryentry{bayesian-optimization}{
    name={Bayesian Optimization},
    text={Bayesian optimization},
    description={is a black-box optimization method used to optimize the parameters of target functions}
}

\newglossaryentry{machine-learning}{
    name={Machine Learning},
    text={machine learning},
    description={is a subfield of artificial intelligence that uses statistical techniques to give computer systems the ability to learn from data without being explicitly programmed}
}

\newglossaryentry{lstm}{
    name={LSTM},
    text={LSTM},
